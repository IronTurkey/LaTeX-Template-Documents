% Defines Chapter heading numbering to roman--------------------------------------------------------------------
\renewcommand{\thechapter}{\Roman{chapter}} % Defines numbering for chapters as roman
\renewcommand{\theHchapter}{\Roman{chapter}}

\makeatletter
\def\@makechapterhead#1{%
\vspace*{\chaptertopskip}%
{\parindent \z@ \raggedright \normalfont
    \ifnum \c@secnumdepth >\m@ne
    \if@mainmatter
        \huge\bfseries Roman Chapter\space \thechapter:\
%        \par\nobreak
%        \vskip 20\p@
    \fi
    \fi
    \interlinepenalty\@M
    \Huge \bfseries #1\par\nobreak
    \vskip \chapterbottomskip
}}
\makeatother
